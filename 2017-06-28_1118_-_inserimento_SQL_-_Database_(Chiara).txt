drop database if exists gopoleis;

create database gopoleis;

use gopoleis;

create table Image(
	code INT NOT NULL AUTO_INCREMENT primary key,
	path varchar (256) NOT NULL,
	filename varchar (100) NOT NULL);

create table Category(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL);

create table Rarity(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL);

create table Role(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (30) NOT NULL);

create table Region(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (50) NOT NULL);

create table Province(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (50) NOT NULL,
	coordinates int NOT NULL,
	zoom int not NULL,
	region int NOT NULL);

create table Medal(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL,
	category int NOT NULL,
	num INT NOT NULL,
	image int NOT NULL);

create table RegionMedal(
	medal INT NOT NULL primary key,
	region INT NOT NULL);

create table HistoricalperiodMedal(
	medal INT NOT NULL primary key,
	historicalperiod INT NOT NULL);

create table StructuretypeMedal(
	medal INT NOT NULL primary key,
	structuretype INT NOT NULL);

create table Card(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL,
	image int NOT NULL,
	description varchar (999) NOT NULL,
	cardvalue int NOT NULL,
	destructionvalue int NOT NULL,
	cost int NOT NULL,
	rarity int NOT NULL);

create table User(
	code INT NOT NULL AUTO_INCREMENT primary key,
	email varchar (100) NOT NULL,
	password varchar (100) NOT NULL,
	enabled ENUM('0', '1') DEFAULT null,
	role int NOT NULL);
		
create table Admin(
	code INT NOT NULL AUTO_INCREMENT primary key,
	user int NOT NULL);
	
create table Organization(
	code INT NOT NULL AUTO_INCREMENT primary key,
	user int NOT NULL,
	admin int DEFAULT null,
	name varchar (100) NOT NULL);

create table Operator(
	code INT NOT NULL AUTO_INCREMENT primary key,
	user int NOT NULL,
	organization int NOT NULL);

create table Coordinates(
	code INT NOT NULL AUTO_INCREMENT primary key,
	latitude varchar (12) NOT NULL,
	longitude varchar (12) NOT NULL);

create table Historicalperiod(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (60) NOT NULL);

create table Structuretype(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (60) NOT NULL);


create table Heritage(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL,
	description varchar (999) NOT NULL,
	image int NOT NULL,
	g1 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	g2 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	g3 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	g4 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	coordinates int NOT NULL,
	province int NOT NULL,
	historicalperiod int NOT NULL,
	structuretype int NOT NULL,
	operator int NOT NULL);

create table Siteinformation(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	description varchar (999) NOT NULL,
	image int NOT NULL,
	heritage int NOT NULL);

create table Treasure(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	description varchar (999) NOT NULL,
	coordinates int NOT NULL,
	heritage int NOT NULL);

create table Enciclopedicpage(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	description varchar (999) NOT NULL,
	curiosity varchar (999) NOT NULL,
	image int NOT NULL,
	heritage int NOT NULL);

create table Question(
	code INT NOT NULL AUTO_INCREMENT primary key,
	question varchar (999) NOT NULL,
	hintonsite varchar (200) NOT NULL,
	hintbypaying varchar (200) NOT NULL,
	answer varchar (100)NOT NULL);

create table Path(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	enabled ENUM('false', 'true') NOT NULL DEFAULT 'false',
	heritage int NOT NULL);

create table Stage(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	curiosity varchar (200) NOT NULL,
	question int NOT NULL,
	coordinates int NOT NULL,
	path int NOT NULL,
	isfinal ENUM('false', 'true') NOT NULL DEFAULT 'false');

create table Review(
	player varchar (100) NOT NULL,
	heritage int NOT NULL,
	likes int NOT NULL DEFAULT 0,
	dislikes int NOT NULL DEFAULT 0,
	review varchar (999),
	primary key (player,heritage));

create table Shape(
	code INT NOT NULL AUTO_INCREMENT primary key,
	type varchar (30) NOT NULL);

create table Validityarea(
	code INT NOT NULL AUTO_INCREMENT primary key,
	heritage int NOT NULL,
	shape int NOT NULL);

create table Polygon(
	code INT NOT NULL AUTO_INCREMENT primary key,
	validityarea int NOT NULL);

create table Vertex(
	code INT NOT NULL AUTO_INCREMENT primary key,
	polygon int NOT NULL,
	coordinates int NOT NULL,
	numvertex int NOT NULL);

create table Circle(
	code INT NOT NULL AUTO_INCREMENT primary key,
	validityarea int NOT NULL,
	center int NOT NULL,
	radius double NOT NULL);

create table Player(
	email varchar (100) NOT NULL primary key,
	title varchar (100),
	coins int NOT NULL DEFAULT 0);

create table PlayerEP(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	viewed ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerSI(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	viewed ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerPath(
	player varchar (100) NOT NULL,
	path int NOT NULL,
	primary key (player, path));

create table PlayerStage(
	player varchar (100) NOT NULL,
	stage int NOT NULL,
	primary key (player, stage));

create table PlayerCard(
	player varchar (100) NOT NULL,
	card int NOT NULL,
	primary key (player, card));

create table PlayerTreasure(
	player varchar (100) NOT NULL,
	treasure int NOT NULL,
	primary key (player, treasure));

create table PlayerHeritage(
	player varchar (100) NOT NULL,
	heritage int NOT NULL,
	primary key (player, heritage));

create table PlayerMedal(
	player varchar (100) NOT NULL,
	medal int NOT NULL,
	primary key (player, medal));

create table Mission(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (100) NOT NULL,
	num int NOT NULL,
	missiontype int NOT NULL,
	granularitylevel int NOT NULL);

create table Missiontype(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL);
	
create table Granularitylevel(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (20) NOT NULL);

create table RegionalMission(
	mission INT NOT NULL primary key,
	region int NOT NULL);

create table ProvincialMission(
	mission INT NOT NULL primary key,
	province int NOT NULL);

create table HeritageMission(
	mission INT NOT NULL primary key,
	heritage int NOT NULL);

create table RarityMission(
	mission INT NOT NULL primary key,
	rarity int NOT NULL);

create table CategoryMission(
	mission INT NOT NULL primary key,
	category int NOT NULL);

create table RegionMedalMission(
	mission INT NOT NULL primary key,
	regionmedal int NOT NULL);

create table HPMedalMission(
	mission INT NOT NULL primary key,
	hpmedal int NOT NULL);

create table ToSMedalMission(
	mission INT NOT NULL primary key,
	tosmedal int NOT NULL);

create table HeritageMedal(
	heritage INT NOT NULL,
	medal INT NOT NULL,
	primary key (heritage, medal));
