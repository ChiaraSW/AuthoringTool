drop database if exists gopoleis;

create database gopoleis;

use gopoleis;

create table Image(
	code INT NOT NULL AUTO_INCREMENT primary key,
	path varchar (256) NOT NULL,
	filename varchar (100) NOT NULL);

create table Category(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL);

create table Rarity(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL);

create table Role(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (30) NOT NULL);

create table Region(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (50) NOT NULL);

create table Province(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (50) NOT NULL,
	coordinates int NOT NULL,
	zoom int not NULL,
	region int NOT NULL);

create table Medal(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100)NOT NULL,
	category int NOT NULL,
	num int NOT NULL,
	image int NOT NULL);

create table Card(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL,
	image int NOT NULL,
	description varchar (999) NOT NULL,
	cardvalue int NOT NULL,
	destructionvalue int NOT NULL,
	cost int NOT NULL,
	rarity int NOT NULL);

create table User(
	code INT NOT NULL AUTO_INCREMENT primary key,
	email varchar (100) NOT NULL,
	password varchar (100) NOT NULL,
	enabled ENUM('0', '1') DEFAULT null,
	role int NOT NULL);	
		
create table Admin(
	code INT NOT NULL AUTO_INCREMENT primary key,
	user int NOT NULL);		
	
create table Organization(
	code INT NOT NULL AUTO_INCREMENT primary key,
	user int NOT NULL,
	admin int DEFAULT null,
	name varchar (100) NOT NULL);		

create table Operator(
	code INT NOT NULL AUTO_INCREMENT primary key,
	user int NOT NULL,
	organization int NOT NULL);

create table Coordinates(
	code INT NOT NULL AUTO_INCREMENT primary key,
	latitude varchar (12) NOT NULL,
	longitude varchar (12) NOT NULL);

create table Historicalperiod(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (60) NOT NULL);

create table Structuretype(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (60) NOT NULL);	


create table Heritage(
	code INT NOT NULL AUTO_INCREMENT primary key,
	name varchar (100) NOT NULL,
	description varchar (999) NOT NULL,
	image int NOT NULL,
	g1 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	g2 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	g3 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	g4 ENUM('false', 'true') NOT NULL DEFAULT 'false',
	coordinates int NOT NULL,
	province int NOT NULL,
	historicalperiod int NOT NULL,
	structuretype int NOT NULL,
	operator int NOT NULL);	

create table Siteinformation(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	description varchar (999) NOT NULL,
	image int NOT NULL,
	heritage int NOT NULL);	

create table Treasure(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	description varchar (999) NOT NULL,
	coordinates int NOT NULL,
	heritage int NOT NULL);	

create table Enciclopedicpage(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	description varchar (999) NOT NULL,
	curiosity varchar (999) NOT NULL,
	image int NOT NULL,
	heritage int NOT NULL);	

create table Question(
	code INT NOT NULL AUTO_INCREMENT primary key,
	question varchar (999) NOT NULL,
	hintonsite varchar (200) NOT NULL,
	hintbypaying varchar (200) NOT NULL,
	answer varchar (100)NOT NULL);

create table Path(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	question int NOT NULL,
	enabled ENUM('false', 'true') NOT NULL DEFAULT 'false',
	heritage int NOT NULL);

create table Stage(
	code INT NOT NULL AUTO_INCREMENT primary key,
	title varchar (150) NOT NULL,
	curiosity varchar (200) NOT NULL,
	question int NOT NULL,
	coordinates int NOT NULL,
	path int NOT NULL);

create table Review(
	player varchar (100) NOT NULL,
	heritage int NOT NULL,
	likes int NOT NULL DEFAULT 0,
	dislikes int NOT NULL DEFAULT 0,
	review varchar (999),
	primary key (player,heritage));

create table Player(
	id varchar (100) NOT NULL primary key,
	username varchar (100) NOT NULL,
	coins int NOT NULL DEFAULT 0);

create table PlayerEP(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	viewed ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerSI(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	viewed ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerPath(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	completed ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerStage(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	solved ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerCard(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	owned ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));

create table PlayerTreasure(
	id varchar (100) NOT NULL,
	code int NOT NULL,
	found ENUM('false', 'true') NOT NULL DEFAULT 'false',
	primary key (id, code));


alter table Medal
add constraint fk1 foreign key Medal (category) references Category (code) ON DELETE CASCADE,
add constraint fk2 foreign key Medal (image) references Image (code) ON DELETE CASCADE;

alter table Card
add constraint fk3 foreign key Card (image) references Image (code) ON DELETE CASCADE,
add constraint fk4 foreign key Card (rarity) references Rarity (code) ON DELETE CASCADE;

alter table User
add constraint fk5 foreign key User (role) references Role (code) ON DELETE CASCADE;

alter table Admin
add constraint fk6 foreign key Admin (user) references User (code) ON DELETE CASCADE;

alter table Organization
add constraint fk7 foreign key Organization (user) references User (code) ON DELETE CASCADE,
add constraint fk8 foreign key Organization (admin) references Admin (code) ON DELETE CASCADE;

alter table Operator
add constraint fk9 foreign key Operator (user) references User (code) ON DELETE CASCADE,
add constraint fk10 foreign key Operator (organization) references Organization (code) ON DELETE CASCADE;

alter table Heritage
add constraint fk11 foreign key Heritage (image) references Image (code) ON DELETE CASCADE,
add constraint fk12 foreign key Heritage (coordinates) references Coordinates (code) ON DELETE CASCADE,
add constraint fk13 foreign key Heritage (province) references Province (code) ON DELETE CASCADE,
add constraint fk14 foreign key Heritage (historicalperiod) references Historicalperiod (code) ON DELETE CASCADE,
add constraint fk15 foreign key Heritage (structuretype) references Structuretype (code) ON DELETE CASCADE,
add constraint fk16 foreign key Heritage (operator) references Operator (code) ON DELETE CASCADE;

alter table Province
add constraint fk17 foreign key Province (region) references Region (code) ON DELETE CASCADE,
add constraint fk18 foreign key Province (coordinates) references Coordinates (code) ON DELETE CASCADE;

alter table Siteinformation
add constraint fk19 foreign key Siteinformation (image) references Image (code) ON DELETE CASCADE,
add constraint fk20 foreign key Siteinformation (heritage) references Heritage (code) ON DELETE CASCADE;

alter table Treasure
add constraint fk21 foreign key Treasure (coordinates) references Coordinates (code) ON DELETE CASCADE,
add constraint fk22 foreign key Treasure (heritage) references Heritage (code) ON DELETE CASCADE;

alter table Enciclopedicpage
add constraint fk23 foreign key Enciclopedicpage (image) references Image (code) ON DELETE CASCADE,
add constraint fk24 foreign key Enciclopedicpage (heritage) references Heritage (code) ON DELETE CASCADE;

alter table Path
add constraint fk25 foreign key Path (question) references Question (code) ON DELETE CASCADE,
add constraint fk26 foreign key Path (heritage) references Heritage (code) ON DELETE CASCADE;

alter table Stage
add constraint fk27 foreign key Stage (question) references Question (code) ON DELETE CASCADE,
add constraint fk28 foreign key Stage (coordinates) references Coordinates (code) ON DELETE CASCADE,
add constraint fk29 foreign key Stage (path) references Path (code) ON DELETE CASCADE;

alter table Review
add constraint fk30 foreign key Review (heritage) references Heritage (code) ON DELETE CASCADE,
add constraint fk31 foreign key Review (player) references Player (id) ON DELETE CASCADE;

alter table PlayerEP
add constraint fk32 foreign key PlayerEP (id) references Player (id) ON DELETE CASCADE,
add constraint fk33 foreign key PlayerEP (code) references Enciclopedicpage (code) ON DELETE CASCADE;

alter table PlayerSI
add constraint fk34 foreign key PlayerSI (id) references Player (id) ON DELETE CASCADE,
add constraint fk35 foreign key PlayerSI (code) references Siteinformation (code) ON DELETE CASCADE;

alter table PlayerPath
add constraint fk36 foreign key PlayerPath (id) references Player (id) ON DELETE CASCADE,
add constraint fk37 foreign key PlayerPath (code) references Path (code) ON DELETE CASCADE;

alter table PlayerStage
add constraint fk38 foreign key PlayerStage (id) references Player (id) ON DELETE CASCADE,
add constraint fk39 foreign key PlayerStage (code) references Stage (code) ON DELETE CASCADE;

alter table PlayerCard
add constraint fk40 foreign key PlayerCard (id) references Player (id) ON DELETE CASCADE,
add constraint fk41 foreign key PlayerCard (code) references Card (code) ON DELETE CASCADE;

alter table PlayerTreasure
add constraint fk42 foreign key PlayerCard (id) references Player (id) ON DELETE CASCADE,
add constraint fk43 foreign key PlayerCard (code) references Treasure (code) ON DELETE CASCADE;
